    function findUniq(arr) {
      let num = arr[0];
      let n = [];
      let s = 0;
      for (let i = 1; i < arr.length; i++) {
          if (num == arr[i]){
              continue;
          } else if (num != arr[i]) {
              n[s] = arr[i];
              s++;
          }
      }
      if (n.length > 1) {
          return num;
      } else if (n.length == 1) {
          return +n;
      }
    }
